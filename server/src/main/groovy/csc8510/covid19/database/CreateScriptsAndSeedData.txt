(
    PATIENT_ID  INTEGER,
    FIRST_NAME  NVARCHAR(200),
    LAST_NAME   NVARCHAR(200),
    AGE         INT,
    GENDER      NVARCHAR(20),
    QUANTITY    DECIMAL,

);
